#include <stdio.h>

int main(void)
{
    int input;
    scanf("%d", &input);

    int array[100];  // 충분한 크기의 배열 확보 (예: 입력이 100 이하인 경우)

    for (int i = 0; i < input; i++)
    {
        scanf("%d", &array[i]);
    }

    int min = 0, max = 0;
    for (int j = 0; j < input; j++)
    {
        if (array[min] > array[j])
            min = j;
        if (array[max] < array[j])
            max = j;
    }

    printf("%d %d", array[min], array[max]);
}

#include <stdio.h>

// 포인터를 사용해 두 정수의 값을 교환하는 함수
void swap(int *x, int *y) {
    int temp = *x;
    *x = *y;
    *y = temp;
}

int main(void) {
    int a, b;
    scanf("%d %d", &a, &b);
    
    swap(&a, &b);  // 주소를 전달하여 값을 교환

    printf("%d %d\n", a, b);  // 교환된 값 출력

    return 0;
}

#include <stdio.h>

int main(void) {
    int n, scores[100], sum = 0, count = 0;
    scanf("%d", &n);

    for (int i = 0; i < n; i++) {
        scanf("%d", &scores[i]);
        sum += scores[i];
    }

    double avg = (double)sum / n;

    for (int i = 0; i < n; i++) {
        if (scores[i] >= avg) {
            count++;
        }
    }

    printf("%d\n", count);
    return 0;
}

#include <stdio.h>

int reverse(int num) {
    int reversed = 0;
    while (num > 0) {
        reversed = reversed * 10 + (num % 10);
        num /= 10;
    }
    return reversed;
}

int main(void) {
    int a, b;
    scanf("%d %d", &a, &b);

    // 두 숫자를 뒤집고 더 큰 값을 출력
    int revA = reverse(a);
    int revB = reverse(b);

    if (revA > revB) {
        printf("%d\n", revA);
    } else {
        printf("%d\n", revB);
    }

    return 0;
}
